results.duration : 0 }); } function reportSuccess(results) { successCount++; tests.push({ "name": currentTestName, "result": true, "message": '' + results.passed + ' passing of ' + results.total + ' tests', "duration": results.duration }); } var startTime = new Date(); function reportAllResults() { var duration = (new Date() - startTime) / 1000; window.global_test_results = { "passed": successCount, "failed": failedCount, "total": successCount + failedCount, "duration": duration, "tests": tests } } function collate() { var results = window.testframe.contentWindow.global_test_results; if (!results) { reportFailure(null); } else if (results.failed > 0) { reportFailure(results); } else { reportSuccess(results); } runNextTest(); } var testIndex = 0; function runNextTest() { if (testIndex >= files.length) { reportAllResults(); return; } currentTestName = files[testIndex]; testIndex++; if (window.testframe) { document.body.removeChild(window.testframe); } var testframe = document.createElement("iframe"); testframe.style.width = "100%"; testframe.style.height = "100%"; testframe.style.position = "absolute"; testframe.style.left = 0; testframe.style.top = 0; testframe.src = currentTestName; document.body.appendChild(testframe); testframe.onload = function() { var maxChecks = 40; var totalChecks = 0; var checker = function() { totalChecks++; if (window.testframe.contentWindow.global_test_results || (totalChecks >= maxChecks)) { collate(); } else { window.setTimeout(checker, 500); } } checker(); } window.testframe = testframe; } runNextTest();
